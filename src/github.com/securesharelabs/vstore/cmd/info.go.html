    33  var infoCmd = &cobra.Command{
    34  	Use:   "info",
    35  	Short: "Print the current node's vStore information (State)",
    36  	Long: `Print the current node's vStore information including:
    37  
    38    - The latest block height ; and
    39    - The total number of transactions stored ; and
    40    - The application merkle roots to create the state Hash.
    41  
    42    The information returned with this command is necessary to perform
    43    the verification of integrity on vStore state instances.
    44  `,
    45  	Run: func(cmd *cobra.Command, args []string) {
    46  
    47  		// Prepare the local RPC client
    48  		// Note: A node must be running in the background
    49  		// TODO: Permit overwrite of RPC remote address
    50  		logger := cmtlog.NewTMLogger(cmtlog.NewSyncWriter(os.Stdout))
    51  		cli, err := rpc.New("http://localhost:26657", "/websocket")
    52  		if err != nil {
    53  			log.Fatalf("could not connect to RPC server: %v", err)
    54  		}
    55  		cli.SetLogger(logger)
    56  
    57  		// Broadcast the transaction
    58  		response, err := cli.ABCIInfo(cmd.Context())
    59  		if err != nil {
    60  			log.Fatalf("could not retrieve ABCI information: %v", err)
    61  		}
    62  
    63  		var state vfs.State
    64  		err = json.Unmarshal([]byte(response.Response.Data), &state)
    65  		if err != nil {
    66  			log.Fatalf("could not parse State JSON from RPC: %v", err)
    67  		}
    68  
    69  		appInfo := struct {
    70  			ABCIVersion  string
    71  			AppVersion   uint64
    72  			LastHeight   int64
    73  			Transactions int64
    74  			MerkleRoots  int64
    75  			AppHash      string
    76  		}{
    77  			response.Response.Version,
    78  			response.Response.AppVersion,
    79  			state.Height,
    80  			state.NumTransactions,
    81  			int64(len(state.MerkleRoots)),
    82  			fmt.Sprintf("%x", response.Response.LastBlockAppHash),
    83  		}
    84  
    85  		if printAsJSON {
    86  			json, _ := json.MarshalIndent(appInfo, "", "  ")
    87  			fmt.Print(string(json) + "\n")
    88  			return // Job done.
    89  		}
    90  
    91  		fmt.Printf("vStore v1.0 (vfs v%d) - ABCI: \n", vfs.AppVersion)
    92  		fmt.Printf("  ABCI Version: %s\n", appInfo.ABCIVersion)
    93  		fmt.Printf("   App Version: %d\n", appInfo.AppVersion)
    94  		fmt.Printf("   Last Height: %d\n", appInfo.LastHeight)
    95  		fmt.Printf("  Transactions: %d\n", appInfo.Transactions)
    96  		fmt.Printf("  Merkle Roots: %d\n", appInfo.MerkleRoots)
    97  		fmt.Printf("      App Hash: %s\n", appInfo.AppHash)
    98  	},
