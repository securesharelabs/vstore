    56  var queryCmd = &cobra.Command{
    57  	Use:   "query",
    58  	Short: "Query your vStore instance for transactions",
    59  	Long: `Query your vStore instance for transactions using:
    60  
    61  	- the transaction hash as returned by the factory subcommand ; or
    62  	- the block height in which the transaction was included ; or
    63  	- the signer public key attached to the transaction.`,
    64  
    65  	Example: `  vstore query
    66    vstore query --hash "XXX"`,
    67  
    68  	Run: func(cmd *cobra.Command, args []string) {
    69  
    70  		// Prepare the local RPC client
    71  		// Note: A node must be running in the background
    72  		// TODO: Permit overwrite of RPC remote address
    73  		logger := cmtlog.NewTMLogger(cmtlog.NewSyncWriter(os.Stdout))
    74  		cli, err := rpc.New("http://localhost:26657", "/websocket")
    75  		if err != nil {
    76  			log.Fatalf("could not connect to RPC server: %v", err)
    77  		}
    78  		cli.SetLogger(logger)
    79  
    80  		// Ask for hash if not provided with --hash
    81  		// TODO: Permit using height or pubkey indexes
    82  		if len(transactionHash) == 0 {
    83  			fmt.Printf("Enter the transaction hash: ")
    84  			reader := bufio.NewReader(os.Stdin)
    85  			input, err := reader.ReadString('\n')
    86  			if err != nil {
    87  				log.Fatalf("could not read transaction hash: %v", err)
    88  			}
    89  
    90  			transactionHash = strings.TrimSuffix(input, "\n")
    91  		}
    92  
    93  		// Parse transaction hash (for query key)
    94  		hbz, err := hex.DecodeString(transactionHash)
    95  		if err != nil {
    96  			log.Fatalf("could not use provided transaction hash: %v", err)
    97  		}
    98  
    99  		// Execute query using RPC client
   100  		response, err := cli.ABCIQuery(cmd.Context(), "/hash", hbz)
   101  
   102  		if err != nil || response.Response.Code != vfs.CodeTypeOK {
   103  			log.Fatalf("error occured on query: (%d - %s) with error: %v", response.Response.Code, response.Response.Log, err)
   104  		}
   105  
   106  		if len(response.Response.Value) == 0 {
   107  			log.Fatalf("could not find transaction with hash: %x", hbz)
   108  		}
   109  
   110  		tx := new(vfsp2p.Transaction)
   111  		err = proto.Unmarshal(response.Response.Value, tx)
   112  		if err != nil {
   113  			log.Fatalf("could not parse Transaction bytes: %v", err)
   114  		}
   115  
   116  		txBody := string(tx.Body)
   117  		if !printDataAsText {
   118  			txBody = fmt.Sprintf("%x", tx.Body)
   119  		}
   120  
   121  		txInfo := struct {
   122  			Signer    string
   123  			Signature string
   124  			Size      int64
   125  			Data      string
   126  		}{
   127  			fmt.Sprintf("%x", tx.Signer.GetEd25519()),
   128  			fmt.Sprintf("%x", tx.Signature),
   129  			int64(tx.Len),
   130  			txBody,
   131  		}
   132  
   133  		if printAsJSON {
   134  			json, _ := json.MarshalIndent(txInfo, "", "  ")
   135  			fmt.Print(string(json) + "\n")
   136  			return // Job done.
   137  		}
   138  
   139  		fmt.Printf("vStore v1.0 (vfs v%d) - ABCI: \n", vfs.AppVersion)
   140  		fmt.Printf("  Signer PubKey: %s\n", txInfo.Signer)
   141  		fmt.Printf("      Signature: %s\n", txInfo.Signature)
   142  		fmt.Printf("           Size: %d\n", txInfo.Size)
   143  		fmt.Printf("           Data: %s\n", txInfo.Data)
   144  	},
