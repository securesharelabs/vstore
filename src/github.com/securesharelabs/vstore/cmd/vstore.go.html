   150  // home directory as provided with homeDir. A teardown function is returned
   151  // as the third return value, you can defer the call to safely close the db.
   152  func openDatabase(name, homeDir string) (cmtdb.DB, string, func(), error) {
   153  	dbPath := filepath.Join(homeDir, "leveldb")
   154  	dbType := cmtdb.BackendType("goleveldb")
   155  
   156  	db, err := cmtdb.NewDB(name, dbType, dbPath)
   157  	if err != nil {
   158  		return nil, dbPath, func() {}, err
   159  	}
   160  
   161  	return db, dbPath, func() {
   162  		if err := db.Close(); err != nil {
   163  			log.Fatalf("error trying to close database: %v", err)
   164  		}
   165  	}, nil
