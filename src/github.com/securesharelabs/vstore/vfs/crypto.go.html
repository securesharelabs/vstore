   117  // Open reads an AES encrypted file (base64-encoded) and decrypts
   118  // its content using a salted password hash. This function expects
   119  // the random salt to be prepended to the ciphertext (8 bytes).
   120  // Open implements SecretProvider
   121  func (id identityFile) Open() ([]byte, error) {
   122  	if len(id.pw) == 0 {
   123  		return []byte{}, errors.New("password must not be empty")
   124  	}
   125  
   126  	// Read the AES ciphertext bytes from file
   127  	// Note: the first 8-bytes contain the random salt
   128  	ctbz, err := id.Bytes()
   129  	if err != nil {
   130  		return []byte{}, err
   131  	}
   132  
   133  	// Extract salt 8-bytes before ciphertext
   134  	salt, ctbz := ctbz[:8], ctbz[8:]
   135  
   136  	// Generate secret from password
   137  	secret, _ := MustGenerateSecret(id.pw, salt)
   138  
   139  	// Decrypt the ciphertext (private key bytes)
   140  	pbz, err := Decrypt(secret, ctbz)
   141  	if err != nil {
   142  		return []byte{}, err
   143  	}
   144  
   145  	return pbz, nil
   146  }

   222  // GenerateSecret generates a 32-bytes secret by creating a SHA-256
   223  // hash of a salted password using a random salt of 8-bytes. If a non-empty
   224  // salt is provided, it is expected to be of 8-bytes length.
   225  // It returns the 32-bytes secret and an 8-bytes salt.
   226  func GenerateSecret(pw, salt []byte) ([]byte, []byte, error) {
   227  	if len(pw) == 0 {
   228  		return []byte{}, []byte{}, errors.New("password must not be empty")
   229  	}
   230  
   231  	if len(salt) == 0 {
   232  		// Generate random 8-bytes salt
   233  		salt = make([]byte, 8)
   234  		if _, err := io.ReadFull(rand.Reader, salt); err != nil {
   235  			return []byte{}, []byte{}, err
   236  		}
   237  	} else if len(salt) != 8 {
   238  		return []byte{}, []byte{}, fmt.Errorf("invalid salt size, want: %d, got: %d", 8, len(salt))
   239  	}
   240  
   241  	// Secret is: SHA256(salt || password)
   242  	var sbuf bytes.Buffer
   243  	sbuf.Grow(8 + len(pw))
   244  	sbuf.Write(salt) // 8-bytes salt
   245  	sbuf.Write(pw)   // password
   246  	secret := tmhash.Sum(sbuf.Bytes())
   247  
   248  	return secret, salt, nil
   249  }

   251  // Encrypt encrypts a plaintext using the secret with the AES block cipher algo.
   252  func Encrypt(secret []byte, data []byte) ([]byte, error) {
   253  	// Prepare AES block cipher
   254  	block, err := aes.NewCipher(secret)
   255  	if err != nil {
   256  		return []byte{}, err
   257  	}
   258  
   259  	gcm, err := cipher.NewGCM(block)
   260  	if err != nil {
   261  		return []byte{}, err
   262  	}
   263  
   264  	// Generate random salt
   265  	salt := make([]byte, gcm.NonceSize())
   266  	if _, err := io.ReadFull(rand.Reader, salt); err != nil {
   267  		return []byte{}, err
   268  	}
   269  
   270  	// Encrypt private key using AES and store base64
   271  	ctbz := gcm.Seal(salt, salt, data, nil)
   272  	return ctbz, nil
   273  }

   275  // Decrypt decrypts a ciphertext using the secret with the AES block cipher algo.
   276  func Decrypt(secret []byte, ciphertext []byte) ([]byte, error) {
   277  	// Prepare AES block cipher
   278  	block, err := aes.NewCipher(secret)
   279  	if err != nil {
   280  		return []byte{}, err
   281  	}
   282  
   283  	gcm, err := cipher.NewGCM(block)
   284  	if err != nil {
   285  		return []byte{}, err
   286  	}
   287  
   288  	saltSize := gcm.NonceSize()
   289  	salt, ct := ciphertext[:saltSize], ciphertext[saltSize:]
   290  
   291  	bz, err := gcm.Open(nil, salt, ct, nil)
   292  	if err != nil {
   293  		return []byte{}, err
   294  	}
   295  
   296  	return bz, nil
   297  }

   299  // MustGenerateIdentity generates a new ed25519 private key and saves it to
   300  // the provided idFile file. A password pw is used to encrypt the private key.
   301  // 8 bytes are added in front of the ciphertext which consist of a random salt.
   302  // The created identity file contains a base64-encoded AES ciphertext prefixed
   303  // with a random salt of 8 bytes.
   304  // This function will panic if any errors occur.
   305  func MustGenerateIdentity(idFile string, pw []byte) (string, string) {
   306  	if len(pw) == 0 {
   307  		panic("password must not be empty")
   308  	}
   309  
   310  	idDir := filepath.Dir(idFile)
   311  	if _, err := os.Stat(idDir); err != nil {
   312  		os.MkdirAll(idDir, 0700)
   313  	}
   314  
   315  	// Generate ed25519 private key
   316  	priv := ed25519.GenPrivKey()
   317  
   318  	// Generate random salt and 32-bytes secret for AES
   319  	secret, salt := MustGenerateSecret(pw, []byte{}) // random salt
   320  
   321  	// Encrypt the private key using AES
   322  	ctbz, err := Encrypt(secret, priv.Bytes())
   323  	if err != nil {
   324  		panic(err.Error())
   325  	}
   326  
   327  	// Salt is added in front of ciphertext (starting 8-bytes)
   328  	// The salt must be in plaintext to decrypt with the password.
   329  	ctbz = append(salt, ctbz...)
   330  
   331  	// Write base64-encoded ciphertext to file
   332  	b64 := base64.StdEncoding.EncodeToString(ctbz)
   333  	err = os.WriteFile(idFile, []byte(b64), 0600)
   334  	if err != nil {
   335  		panic(err.Error())
   336  	}
   337  
   338  	// Also *always* create a (cleartext) co-located .pub file
   339  	pubFile := idFile + ".pub"
   340  	b64_pub := base64.StdEncoding.EncodeToString(priv.PubKey().Bytes())
   341  	err = os.WriteFile(pubFile, []byte(b64_pub), 0644)
   342  	if err != nil {
   343  		panic(err.Error())
   344  	}
   345  
   346  	// Returns pair of co-located files
   347  	return idFile, pubFile
   348  }
