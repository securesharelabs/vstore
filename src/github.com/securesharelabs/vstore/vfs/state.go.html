    18  // State describes the vstore application state which consists of a latest
    19  // blockchain height, a total number of transactions and cryptographic
    20  // commitments about transaction data (merkle roots).
    21  type State struct {
    22  	db cmtdb.DB

    24  	// NumTransactions is essentially the total number of transactions processed.

    29  	// MerkleRoots contains the cryptographic commitments for transactions that

    35  // MerkleRoots returns a slice of merkle roots that is *deterministic* due to
    36  // keys always being sorted lexicographically.
    37  func (s State) SortedMerkleRoots() [][]byte {
    38  	// Sort keys first (deterministic)
    39  	max := len(s.MerkleRoots)
    40  	keys := make([]string, max)
    41  	roots := make([][]byte, max)
    42  
    43  	i := 0
    44  	for k := range s.MerkleRoots {
    45  		keys[i] = k
    46  		i++
    47  	}
    48  
    49  	// Sort keys lexicographically
    50  	sort.Strings(keys)
    51  
    52  	// Iterate over *keys* for determinism
    53  	for j, k := range keys {
    54  		v := s.MerkleRoots[k]
    55  		roots[j] = v
    56  	}
    57  
    58  	return roots
    59  }

    61  // Hash returns the hash of the application state. This is computed as the merkle
    62  // root of all the committed transaction hashes using a deterministic merkle root
    63  // slices as produced with MerkleRoots().
    64  // The produced hash can be used to verify the integrity of the State.
    65  // This function is used as the "AppHash"
    66  func (s State) Hash() []byte {
    67  	if len(s.MerkleRoots) == 0 {
    68  		return make([]byte, 32)
    69  	}
    70  
    71  	// Compute merkle root of all committed transactions
    72  	return merkle.HashFromByteSlices(s.SortedMerkleRoots())
    73  }

    90  // loadState reads the state key from the database and tries to unmarshal

    92  func loadState(db cmtdb.DB) State {
    93  	var state State
    94  	state.db = db
    95  	stateBytes, err := db.Get(stateKey)
    96  	if err != nil {
    97  		panic(err)

    99  	if len(stateBytes) == 0 {
