    82  func (p SignedTransaction) ToProto() *vfsp2p.Transaction {
    83  	// Make public key transportable
    84  	pk := cmtp2p.PublicKey{
    85  		Sum: &cmtp2p.PublicKey_Ed25519{
    86  			Ed25519: p.Signer.Bytes(),
    87  		},
    88  	}
    89  
    90  	// Don't create protobuf without hash
    91  	if len(p.Hash) == 0 {
    92  		p.Hash = ComputeHash(&p)
    93  	}
    94  
    95  	tx := new(vfsp2p.Transaction)
    96  	tx.Signer = pk
    97  	tx.Signature = p.Signature
    98  	tx.Hash = p.Hash
    99  	tx.Time = time.Unix(p.Time.Unix(), 0)
   100  	tx.Len = uint32(len(p.Data))
   101  	tx.Body = p.Data
   102  
   103  	return tx

   109  // ComputeHash computes the SHA256 hash of a signed transaction
   110  // The transaction hash consists of a SHA256 of the signer public key,
   111  // followed by the data and the attached timestamp bytes.
   112  func ComputeHash(p *SignedTransaction) []byte {
   113  	psize := ed25519.PubKeySize
   114  
   115  	// Timestamp bytes attached to hashed message
   116  	tzb := make([]byte, 8)
   117  	binary.BigEndian.PutUint64(tzb, uint64(p.Time.Unix()))
   118  
   119  	// Tx hash is: sha256(owner || data || sigtime)
   120  	var hbuf bytes.Buffer
   121  	hbuf.Grow(psize + p.Size + timestampSize)
   122  	hbuf.Write(p.Signer) // adding pubkey
   123  	hbuf.Write(p.Data)   // adding data
   124  	hbuf.Write(tzb)      // adding timestamp
   125  
   126  	return tmhash.Sum(hbuf.Bytes())
   127  }
