    74  // validateTx validates that the bytes slice is not empty, and that the data
    75  // contains at least the 32 bytes of the owner pubkey, 64 bytes of the signature

    77  func (app *VStoreApplication) validateTx(tx []byte) uint32 {
    78  	// Expects valid marshalled format for vfsp2p.Transaction
    79  	stx, err := FromBytes(tx)
    80  	if err != nil {
    81  		return CodeTypeInvalidFormatError
    82  	}
    83  
    84  	if stx.Size == 0 || len(stx.Data) == 0 {
    85  		return CodeTypeEmptyDataError
    86  	}
    87  
    88  	if !stx.Verify() {
    89  		return CodeTypeInvalidSignatureError
    90  	}
    91  
    92  	return CodeTypeOK
    93  }

    95  // processFinalizeBlock processes a slice of transactions from the request,
    96  // the creates signed data payloads and updates the Height and NumTransactions
    97  // as specified in the request.
    98  // An error code will be returned for invalid transactions.
    99  func (app *VStoreApplication) processFinalizeBlock(
   100  	_ context.Context,
   101  	req *abci.RequestFinalizeBlock,
   102  ) []*abci.ExecTxResult {
   103  	respTxs := make([]*abci.ExecTxResult, len(req.Txs))
   104  
   105  	// Reset stages
   106  	app.stage = make([]SignedTransaction, 0)
   107  
   108  	// Stage the block data
   109  	for i, tx := range req.Txs {
   110  		// Extract pubkey (32b), signature (64b), timestamp (8b) and data
   111  		payload, err := NewSignedTransactionFromBytes(tx)
   112  		if err != nil {
   113  			respTxs[i] = &abci.ExecTxResult{
   114  				Code:   CodeTypeInvalidFormatError,
   115  				Data:   payload.Hash,
   116  				Events: []abci.Event{},
   117  			}
   118  
   119  			// This transaction won't be staged!
   120  			continue
   121  		}
   122  
   123  		// Stage this transaction
   124  		app.stage = append(app.stage, *payload)
   125  
   126  		respTxs[i] = &abci.ExecTxResult{
   127  			Code:   CodeTypeOK,
   128  			Data:   payload.Hash,
   129  			Events: []abci.Event{},
   130  		}
   131  
   132  		app.state.NumTransactions++
   133  	}
   134  
   135  	app.state.Height = req.Height
   136  	return respTxs
   137  }

   139  // commitMerkleRoots computes merkle roots per owner public key
   140  // and stores them in the merkleRoots property.
   141  func (app *VStoreApplication) commitMerkleRoots() {
   142  	if len(app.state.MerkleRoots) == 0 {
   143  		app.state.MerkleRoots = make(map[string][]byte, 0)
   144  	}
   145  
   146  	for _, payload := range app.stage {
   147  		pub := payload.PublicKey()
   148  		txs := [][]byte{payload.Hash} // merkle root computed with transaction hash
   149  
   150  		// Prepend merkle root if it exists
   151  		if mr, ok := app.state.MerkleRoots[pub]; ok {
   152  			txs = append([][]byte{mr}, txs...)
   153  		}
   154  
   155  		// Compute merkle root by owner public key
   156  		merkleRoot := merkle.HashFromByteSlices(txs)
   157  		app.state.MerkleRoots[pub] = merkleRoot
   158  	}
   159  }

   185  // addTransactionByHeight appends the transaction hash to
   186  // the block height transaction index.
   187  func (app *VStoreApplication) addTransactionByHeight(tx SignedTransaction) error {
   188  	txes := [][]byte{}
   189  
   190  	// Indexes hashes by height with prefix "vfs:height:block-X"
   191  	heightStr := strconv.FormatInt(app.state.Height, 10) // base10
   192  	dbKey_byHeight := prefixKeyWith([]byte(heightStr), vfsPrefixKeyByHeight)
   193  
   194  	// Do we have hashes indexed by this height already?
   195  	data, err := app.state.db.Get(dbKey_byHeight)
   196  	if err != nil {
   197  		return err
   198  	}
   199  
   200  	if len(data) > 0 {
   201  		json.Unmarshal([]byte(data), &txes)
   202  	}
   203  
   204  	// Adds transaction hash by height
   205  	txes = append(txes, tx.Hash)
   206  	byHeight, _ := json.Marshal(txes)
   207  
   208  	// Stores transaction hash to index
   209  	err = app.state.db.Set(dbKey_byHeight, byHeight)
   210  	return err
   211  }

   240  // readTransactionFromDB fetches a transaction from the database.
   241  // Given a transaction hash, the transaction content will be decrypted,
   242  // otherwise the index is read to retrieve the hash and a second query
   243  // is executed to fetch the transaction content by hash.
   244  func (app *VStoreApplication) readTransactionFromDB(
   245  	queryType string,
   246  	value []byte,
   247  ) ([]byte, error) {
   248  	var (
   249  		queryKey []byte = getQueryKey(queryType, value)
   250  	)
   251  
   252  	// Read from the database
   253  	data, err := app.state.db.Get(queryKey)
   254  	if len(data) == 0 || err != nil {
   255  		return []byte{}, err
   256  	}
   257  
   258  	// TODO: Return array of transaction for height/pubkey indexes
   259  	if queryType != QueryType_Default {
   260  		return []byte{}, nil
   261  	}
   262  
   263  	// Unlock the decryption secret
   264  	secret, err := app.priv.Identity().Secret()
   265  	if err != nil {
   266  		return []byte{}, nil
   267  	}
   268  	defer func() { secret = []byte{} }()
   269  
   270  	// Decrypt the transaction data with the node's secret
   271  	txData, err := Decrypt(secret, data)
   272  	if err != nil {
   273  		return []byte{}, nil
   274  	}
   275  
   276  	return txData, nil
   277  }

   337  // PrepareProposal is called only when the node is a proposer. CometBFT stages
   338  // a set of transactions for the application.
   339  // NOTE: we assume that CometBFT won't provide too many transactions for 1 block.
   340  // PrepareProposal implements abci.Application
   341  func (app *VStoreApplication) PrepareProposal(
   342  	ctx context.Context,
   343  	proposal *abci.RequestPrepareProposal,
   344  ) (*abci.ResponsePrepareProposal, error) {
   345  	// Validate transactions before creating proposal
   346  	blockData := make([][]byte, 0, len(proposal.Txs))
   347  	for _, tx := range proposal.Txs {
   348  		resp, err := app.CheckTx(ctx, &abci.RequestCheckTx{Tx: tx})
   349  		if resp.Code != CodeTypeOK || err != nil {
   350  			continue
   351  		}
   352  
   353  		blockData = append(blockData, tx)
   354  	}
   355  
   356  	// Forwarded block data are all valid transactions
   357  	return &abci.ResponsePrepareProposal{Txs: blockData}, nil
   358  }

   403  // Commit is called after FinalizeBlock and after the CometBFT state updates.
   404  // The vfs application persists the staged data (from FinalizeBlock) in database
   405  // in a modified key-value store where the key is the tx hash, and where
   406  // values describe marshalled protobuf instances of vfsp2p.Transaction.
   407  // Commit implements abci.Application
   408  func (app *VStoreApplication) Commit(
   409  	_ context.Context,
   410  	commit *abci.RequestCommit,
   411  ) (*abci.ResponseCommit, error) {
   412  	// Read the encryption secret
   413  	secret, err := app.priv.Identity().Secret()
   414  	if err != nil {
   415  		return nil, err
   416  	}
   417  
   418  	defer func() {
   419  		secret = []byte{}
   420  	}()
   421  
   422  	// Persist all the staged data in vfs
   423  	for _, payload := range app.stage {
   424  		// Use transaction hash as the key (index by hash)
   425  		dbKey := prefixKey(payload.Hash)
   426  
   427  		// Transaction hash must not exist
   428  		if resp, err := app.state.db.Has(dbKey); err != nil || resp {
   429  			return nil, errors.New("transaction hash already exists")
   430  		}
   431  
   432  		// Encrypt the transaction using the node's secret
   433  		encProto, err := Encrypt(secret, payload.Bytes())
   434  		if err != nil {
   435  			return nil, err
   436  		}
   437  
   438  		// Stores an encrypted vfsp2p.Transaction protobuf payload
   439  		err = app.state.db.Set(dbKey, encProto)
   440  		if err != nil {
   441  			return nil, err
   442  		}
   443  	}
   444  
   445  	// Indexes transaction hash by height and signer pubkey
   446  	app.commitTransactionHashes()
   447  
   448  	// Save the State in database with updated merkle roots
   449  	app.commitStateTransitions()
   450  
   451  	// Response OK
   452  	return &abci.ResponseCommit{}, nil
   453  }

   485  // getQueryKey returns a prefixed database key depending of a queryType.
   486  func getQueryKey(queryType string, value []byte) []byte {
   487  	switch queryType {
